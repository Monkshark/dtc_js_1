자주 쓰는 계산 문법
+, -, *, /

          a = 5 + 7;
          a = a + 5;

          a = 5 + 7; // 12
          a = a + 5; // 17

          a = 5 + 7; // 12
          a += 5 // a = a + 5
        

          let a = 1, b = 2, c = 3, d = 4;
          a += 3;
          b -= 5;
          c *= 3;
          d /= 2;
        

          let a = 1, b = 2, c = 3, d = 4;
          a += 3; // 4
          b -= 5; // -3
          c *= 3; // 9
          d /= 2; // 2
        

          let a = 1;

          a++; // 1
          a; // 2

          ++a; // 3
          a; // 3
        

          let a = 5;

          a-- // 5
          a // 4

          --a // 3
          a // 3
